/**
 * Sistema de tratamento de erros com log e notificação
 * @author Danilo Gomes - Gomes Serviços Ágeis
 * @param {Function} fn - Função a ser executada com proteção
 * @param {string} contexto - Nome do processo para identificação
 */
function executarComSeguranca(fn, contexto) {
  try {
    const inicio = new Date();
    const resultado = fn();
    
    // Log de sucesso com tempo de execução
    const duracao = ((new Date() - inicio) / 1000).toFixed(2);
    Logger.log(`✅ [${contexto}] Concluído em ${duracao}s`);
    
    return { sucesso: true, dados: resultado };
    
  } catch (erro) {
    // Registra erro em planilha de logs
    registrarErroNaPlanilha(contexto, erro);
    
    // Notifica administrador via WhatsApp
    enviarAlertaWhatsApp({
      tipo: 'ERRO_SISTEMA',
      processo: contexto,
      mensagem: erro.message,
      stack: erro.stack
    });
    
    return { sucesso: false, erro: erro.message };
  }
}
